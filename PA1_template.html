weekday <- function(day)
  {
    if(day %in% c("Monday", "Tuesday", "Wednesday","Thursday","Friday")) 
      {
      weekdayorend = 'Weekday'
    } else 
      {
    weekdayorend = 'Weekend'

    }

weekdayorend

## plot time series plot
